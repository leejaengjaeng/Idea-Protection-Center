CREATE TABLE `idea_pc`.`user` (
  `uid` INT NOT NULL AUTO_INCREMENT,
  `id` VARCHAR(45) NOT NULL,
  `pw` VARCHAR(45) NOT NULL,
  `name` VARCHAR(45) NOT NULL,
  `phone` VARCHAR(45) NULL,
  `addr1` VARCHAR(45) NULL,
  `addr2` VARCHAR(45) NULL,
  `addr3` VARCHAR(45) NULL,
  `email` VARCHAR(45) NULL,
  `role` VARCHAR(45) NULL,
  PRIMARY KEY (`uid`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;


CREATE TABLE `idea_pc`.`patientent_lawyer` (
  `lid` INT NOT NULL AUTO_INCREMENT,
  `uid` VARCHAR(45) NOT NULL,
  `license_scan_img` VARCHAR(100) NULL,
  `license_number` VARCHAR(45) NULL,
  PRIMARY KEY (`lid`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;


CREATE TABLE `idea_pc`.`process` (
  `pid` INT NOT NULL AUTO_INCREMENT,
  `start_date` DATETIME NULL DEFAULT CURRENT_TIMESTAMP,
  `answer_date` DATETIME NULL,
  `p_name` VARCHAR(45) NULL,
  `answer_name` VARCHAR(45) NULL,
  `category1` VARCHAR(45) NULL,
  `answer_category1` VARCHAR(45) NULL,
  `category2` VARCHAR(45) NULL,
  `answer_category2` VARCHAR(45) NULL,
  `meaning` VARCHAR(45) NULL,
  `answer_meaning` VARCHAR(45) NULL,
  `start_pid` VARCHAR(45) NULL,
  PRIMARY KEY (`pid`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;


CREATE TABLE `idea_pc`.`process_files` (
  `pfid` INT NOT NULL AUTO_INCREMENT,
  `start_pid` INT NULL,
  `file_path` VARCHAR(100) NULL,
  PRIMARY KEY (`pfid`));
