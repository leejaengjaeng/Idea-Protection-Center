CREATE TABLE `idea_pc`.`user` (
  `uid` INT NOT NULL AUTO_INCREMENT,
  `id` VARCHAR(45) NOT NULL,
  `pw` VARCHAR(45) NOT NULL,
  `name` VARCHAR(45) NOT NULL,
  `phone` VARCHAR(45) NULL,
  `addr1` VARCHAR(45) NULL,
  `addr2` VARCHAR(45) NULL,
  `addr3` VARCHAR(45) NULL,
  `email` VARCHAR(45) NULL,
  `role` VARCHAR(45) NULL,
  `is_member` VARCHAR(45) NULL,
  PRIMARY KEY (`uid`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

ENGINE = InnoDB

CREATE TABLE `registration_patent` (
  `rid` int(11) NOT NULL AUTO_INCREMENT,
  `typeofinvent` varchar(45) DEFAULT NULL,
  `title` varchar(45) DEFAULT NULL,
  `summary` text,
  `whyinvent` text,
  `problem` text,
  `solution` text,
  `effect` text,
  `core_element` text,
  `uid` int(11) DEFAULT NULL,
  `iscomplete` tinyint(4) DEFAULT '0',
  `start_rid` int(11) DEFAULT NULL,
  `re_typeofinvent` varchar(45) DEFAULT NULL,
  `re_title` varchar(45) DEFAULT NULL,
  `re_summary` text,
  `re_whyinvent` text,
  `re_problem` text,
  `re_solution` text,
  `re_effect` text,
  `re_core_element` text,
  `lid` int(11) DEFAULT NULL,
  `condition` varchar(45) DEFAULT NULL,
  `pre_apply_date` datetime DEFAULT NULL,
  `apply_date` datetime DEFAULT NULL,
  `registration_date` datetime DEFAULT NULL,
  PRIMARY KEY (`rid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
