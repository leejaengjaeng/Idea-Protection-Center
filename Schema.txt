CREATE TABLE `user` (
  `uid` int(11) NOT NULL AUTO_INCREMENT,
  `id` varchar(45) NOT NULL,
  `pw` varchar(45) NOT NULL,
  `name` varchar(45) NOT NULL,
  `phone` varchar(45) DEFAULT NULL,
  `addr1` varchar(45) DEFAULT NULL,
  `addr2` varchar(45) DEFAULT NULL,
  `addr3` varchar(45) DEFAULT NULL,
  `email` varchar(45) DEFAULT NULL,
  `role` varchar(45) DEFAULT NULL,
  `is_member` varchar(45) DEFAULT NULL,
  `profileimg` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`uid`)
) ENGINE=InnoDB AUTO_INCREMENT=18 DEFAULT CHARSET=utf8;

  `whyinvent` text,

  `solution` text,
  `effect` text,
  `core_element` text,

  `start_rid` int(11) DEFAULT NULL,
  `prev_rid` int(11) DEFAULT '-1',
  `re_typeofinvent` varchar(45) DEFAULT NULL,
  `re_title` varchar(45) DEFAULT NULL,
  `re_summary` text,
  `re_whyinvent` text,
  `re_problem` text,
  `re_solution` text,
